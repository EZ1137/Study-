```javascript
3 == 3
// true
3 === 3
// true

3 == '3'
// true
3 === '3'
// false
```

```javascript
true && true
// true and true		TRUE
true && false
// true and false		FALSE
false && false
// false and false	FALSE
```

```javascript
true || true
// true or true			TRUE
true || false
// true or false		TRUE
false || false
// false or false		FALSE
```

```javascript
!false
// not false		TRUE
!true
// not true			FALSE
```

```javascript
(7 > 3) && (6 < -1)
// (true) and (false)		FALSE
(7 > 3) || (6 < -1)
// (true) or (false)		TRUE
!(7 > 3)
// not true							FALSE
```

```javascript
var scoreString = '10';
console.log(Number(scoreString));
// 10
console.log(typeof scoreString);
// string
console.log(typeof Number(scoreString));
// number
console.log(typeof 10);
// number
```

```javascript
var iPad = [800, 'Spacegray', true];
console.log(typeof iPad);
// object

var brands = ['Apple', 'Coca-Cola', 'Starbucks'];
console.log(typeof brands);
// object
console.log(brands[1]);
// Coca-Cola

var products = [
  ['iPhone', 'iMac', 'MacBook'], 
  ['Coke', 'Diet Coke'], 
  ['Americano', 'Latte', 'Teavana']
];
console.log(products);
// [Array(3), Array(2), Array(3)]
console.log(products[0]);
// ['iPhone', 'iMac', 'MacBook']
console.log(products[1][1]);
// Diet Coke

var appleProducts = products[0];
console.log(appleProducts[1]);
// iMac
```

```javascript
var person = ['Jay', 29, 'Korea'];
console.log(person[0]);
// Jay

var person = {
  name: 'Jay', 
  age: 29, 
  nationality: 'Korea'
}
console.log(person['name']);
// Jay
console.log(person.name);
// Jay
```
